# 세션 인증 테스트용 curl 명령어들
# 서버가 실행 중일 때 사용하세요

echo "=== 세션 인증 테스트 ==="

# 1. 세션 없이 사용자 프로필 조회 (401 에러 예상)
echo "1. 사용자 프로필 조회 (세션 없음)"
curl -X GET http://localhost:5000/api/v1/users/me

echo -e "\n2. 내 동아리 목록 조회 (세션 없음)"
curl -X GET http://localhost:5000/api/v1/users/me/clubs

echo -e "\n3. 동아리 정보 수정 (세션 없음)"
curl -X PATCH http://localhost:5000/api/v1/clubs/1 \
  -H "Content-Type: application/json" \
  -d '{"name": "테스트 동아리"}'

echo -e "\n4. 지원자 목록 조회 (세션 없음)"
curl -X GET "http://localhost:5000/api/v1/applications?club_id=1"

echo -e "\n5. 동아리 공지 등록 (세션 없음)"
curl -X POST http://localhost:5000/api/v1/clubs/1/notices \
  -H "Content-Type: application/json" \
  -d '{"title": "테스트 공지", "content": "테스트 내용"}'

echo -e "\n=== 테스트 완료 ==="
echo "모든 요청에서 401 에러가 나오면 세션 인증이 정상 작동하는 것입니다!"
